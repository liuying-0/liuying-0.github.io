不定积分与定积分的计算

发表于2023-05-28|更新于2023-07-18|数学高数
|字数总计:1.9k|阅读时长:9分钟|阅读量:

基本积分公式

$$
\begin{flalign*}
& (1)\int0\mathrm{d}x=C\
&(2)\int1\mathrm{d}x=\int\mathrm{d}x=x+C & \
&(3)\int x^{s}\mathrm{d}x={\frac{1}{a+1}}x^{s+1}+C(a\neq-1)\
&(4)\int{\frac{1}{x}}\mathrm{d}x=\ln\mid x\mid+C\
&(5)\int a^{x}\mathrm{d}x={\frac{a^{x}}{\ln a}}+C(a>0,a\neq1)\
&(6)\int e^{x}\mathrm{d}x=e^{x}+C\
&(7)\int\sin x\mathrm{d}x=-\cos x+C\
&(8)\int\cos x\mathrm{d}x=\sin x+C \
&(9)\int\tan x\mathrm{d}x=-\ln|\cos x|+C\
&(10)\int\cot x\mathrm{d}x=\ln|\sin x|+C \
&(11)\int\text{sec}x\text{d}x=\ln\mid\sec x+\tan x\mid+C\
&(12)\int\text{csc}x\text{d}x=\ln\mid\text{csc}x-\cot x\mid+C \
&(13)\int\sec^{2}x\mathrm{d}x=\tan x+C\
&(14)\int\csc^{2}x\mathrm{d}x=-\cot x+C\
&(15)\int{\frac{1}{a^{3}+x^{2}}}\mathrm{d}x={\frac{1}{a}}\operatorname{arctan}{\frac{x}{a}}+C\
&(16)\int{\frac{1}{a^{3}-x^{3}}}\mathrm{d}x={\frac{1}{2a}}\ln\mid{\frac{a+x}{a-x}}+C \
&(17)\int\frac{1}{\sqrt{a^2-x^2}}\mathrm dx=\arcsin\frac{x}{a}+C\
&(18)\int{\frac{\mathrm{d}x}{\sqrt{x^{2}\pm a^{2}}}}=\ln\mid x+{\sqrt{x^{2}\pm a^{2}}}+C\
\end{flalign*}
$$

基本积分方法

凑微分法（也称第一换元法）

设$f(u)$连续,$\varphi(x)$具有连续的一阶导数，则有公式：

$$
\int f[\varphi(x)]\varphi’(x)\mathrm{d}x=\int f[\varphi(x)]\mathrm{d}\varphi(x)\xlongequal{令\varphi(x)=u}\int f(u)\mathrm{d}u\
\xlongequal{如果}F(u)+C\xlongequal{则}F(\varphi(x))+C
$$

换元积分法（也称第二换元法）

设$f(x)$连续,$x=\varphi(t)$具有连续导数$\varphi’(t)$,且$\varphi’(t)\neq 0$,则

$$
\int f(x)\mathrm{d}x\xlongequal{x=\varphi(t)}\big(\int f(\varphi(t))\varphi’(t)\mathrm{d}t\big)_{t=\varphi(x)}
$$

其中右边表示对$t$积分之后再以 $x=\varphi(t)$的反函数$t=\psi(x)$代回成$x$的函数

常见的几种典型类型的换元法


$$
\begin{flalign*}
&以下式子中,R(u,v)表示u,v的有理函数&\
&(1)\int R(x,\sqrt{a^{2}-x^{2}})\mathrm{d}x,\int R(x,\sqrt{x^{2}\pm a^{2}})\mathrm{d}x型,a>0\
&含\sqrt{a^2-x^2},令x= a\sin t,{\rm d}x = a\cos t{\rm d}t\
&含\sqrt{x^2+a^2 }
,令x=a\tan t,{\rm d} x=a\sec^{2} t {\rm d} t\
&含\sqrt{x^2 -a^2},令x= a\sec t,{\rm d}x = a\sec t\tan t{\rm d}t\
&(2)\int R(x,\sqrt[n]{ax+b},\sqrt[m]{ax+b})\mathrm{d}x型,a\neq0.\
&令\sqrt[mn]{ax+b}=t,x=\frac{t^{mn}-b}{a},\mathrm{d}x=\frac{mn}{a}t^{mn-1}\mathrm{d}t\
&(3)\int R\Big(x,\sqrt{\frac{ax+b}{cx+d}}\Big)\mathrm{d}x型\
&令{\sqrt{\frac{a x+b}{c x+d}}}=t,x={\frac{d t^{2}-b}{a-c t^{2}}},\mathrm{d}x={\frac{2(a d-b c)t}{(a-c t^{2})^{2}}}\mathrm{d}t,
其中设 a d-b c\neq0\
&(4)\int R(\sin x，\cos x)dx型\
&命{\tan}{\frac{x}{2}}=t,则{\sin}x={\frac{2t}{1+t^{2}}},{\cos}x={\frac{1-t^{2}}{1+t^{2}}},{\mathrm{d}}x={\frac{2}{1+t^{2}}}{\mathrm{d}}t\
&此称万能代换,这样代换通常会产生复杂的计算,非到不得已时不用
\end{flalign*}
$$

定积分的换元积分法

$$
\begin{flalign*}
&设f(x)在[a,b] 上连续&\
&x = \varphi(t)满足条件:a = \varphi(\alpha), b=\varphi(β)\
&并且当t在以α,β为端点的闭区间I 上变动时, a\le \varphi(t) \le b ,\varphi’(t)连续\
&则有定积分的换元积分公式
\end{flalign*}
$$

$$
\int_a^b f(x)\mathrm{d}x=\int_\alpha^\beta f[\varphi(t)]\varphi’(t)\mathrm{d}t
$$

$$
\begin{flalign*}
&【注】(1)这里 t=\alpha,t=\beta由关系:x=\varphi(t)分别与x=a,x=b对应,\alpha可能小于\beta,也可能大于\beta&\
&(2)换元时，积分上、下限应跟着换，而不必像不定积分那样求出原函数后代回成原变量x\
&(3)定积分换元积分法的定理与方法与不定积分的不一样\
\end{flalign*}
$$

分部积分法

设$u(x),v(x)$均有连续导数，则
$$
\int u(x)v’(x){\rm d}x=u(x)v(x)-\int v(x) u’(x)dx
$$
或

$$
\int u(x)\operatorname{d}\left[v(x)\right]=u(x)v(x)-\int v(x)\operatorname{d}\left[u(x)\right]
$$

以及

$$
\int_{a}^{b}u(x)v^{\prime}(x)\mathrm{d}x=u(x)v(x)\bigg|{a}^{b}-\int{a}^{b}v(x)u^{\prime}(x)\mathrm{d}x
$$

或

$$
\int_{a}^{b}u(x)\mathrm{d}[v(x)]=u(x)v(x)\bigg|{a}^{b}-\int{a}^{b}v(x)\mathrm{d}[u(x)]
$$

$$
\begin{flalign*}
&分部积分法的关键与特点是,将被积函数写成两函数之积,一为 u(x) ,一为已求导的形式 v’(x)&\
&使用此公式后,将 u(x) 转化为求导形式u’(x),而将 v’(x)转化为其原函数 v(x)\
&可见选取 v’(x)(或 {\rm d} v(x))应能积分,这是原则
\end{flalign*}
$$

常见的用分部积分的几种函数类型

$$
\begin{flalign*}
&(1)\int x^n \mathrm{e}^x \mathrm{d} x, \int x^n \sin x \mathrm{d} x, \int x^n \cos x \mathrm{d} x 型, 分别改写为 \int x^n \mathrm{e}^x\mathrm{d} x=\int x^n \mathrm{d}\left(\mathrm{e}^x\right),&\
&\int x^n \sin x \mathrm{d} x=-\int x^n \mathrm{d}(\cos x), \int x^n \cos x \mathrm{d} x=\int x^n \mathrm{d}(\sin x), 使用分部积分公式即可\
&(2) \int x^n \ln x \mathrm{d} x, \int x^n \arctan x \mathrm{d} x, \int x^n \arcsin x \mathrm{d} x 型, 分别改写为\
&\int x^n \ln x \mathrm{d} x=\frac{1}{n+1} \int \ln x \mathrm{d}\left(x^{n+1}\right), \int x^n \arctan x \mathrm{d} x=\frac{1}{n+1} \int \arctan x \mathrm{d}\left(x^{n+1}\right),\
&\int x^n \arcsin x \mathrm{d} x=\frac{1}{n+1} \int \arcsin x \mathrm{d}\left(x^{n+1}\right), 使用分部积分公式即可\
&以上(1)(2)中n均为正整数\
&(3) \int \mathrm{e}^x \sin x \mathrm{d} x, \int \mathrm{e}^x \cos x \mathrm{d} x 型,应连用两次,移项解方程可得\
&必须指出,分部积分法的使用对象,远远不止这几种,其优先级一般按照【反对幂指三】来排列，\
&其中三角函数优先级最高，即优先作为v’(x)来分部积分
\end{flalign*}
$$

几个十分有用的定积分公式

$$
\begin{flalign*}
&(1) \int_0^a \sqrt{a^2-x^2} \mathrm{d} x=\frac{1}{4} \pi a^2, \int_{-a}^a \sqrt{a^2-x^2} \mathrm{d} x=\frac{1}{2} \pi a^2 (半圆面积)&
\end{flalign*}
$$
$$
\begin{flalign*}
&(2) 设 f(x) 在 -a, a 上是个连续的偶函数, 则&
\end{flalign*}
$$

$$
\int_{-a}^a f(x) \mathrm{d} x=2 \int_0^a f(x) \mathrm{d} x
$$
$$
\begin{flalign*}
&(3) 设 f(x) 在 -a, a 上是个连续的奇函数, 则 \int_{-a}^a f(x) \mathrm{d} x=0.&
\end{flalign*}
$$
$$
\begin{flalign*}
&(4) 设 f(x) 在 (-\infty,+\infty) 内是以 T 为周期的连续函数, 则对于任意的常数 a, 恒有&
\end{flalign*}
$$
$$
\int_a^{a+T} f(x) \mathrm{d} x=\int_0^T f(x) \mathrm{d} x . \quad \int_a^{a+n T} f(x) \mathrm{d} x=n \int_0^T f(x) \mathrm{d} x(n \in \mathbf{N}) .
$$
$$
\begin{flalign*}
&(5) 华里士公式:&
\end{flalign*}
$$
$$
\begin{aligned}
\int_0^{\frac{\pi}{2}} \sin ^n x \mathrm{d} x & =\int_0^{\frac{\pi}{2}} \cos ^n x \mathrm{d} x \
& = \begin{cases}\frac{n-1}{n} \cdot \frac{n-3}{n-2} \cdots \cdots \frac{1}{2} \cdot \frac{\pi}{2}, \text { 当 } n \text { 为正偶数, } I_0=\frac{\pi}{2} . \
\frac{n-1}{n} \cdot \frac{n-3}{n-2} \cdots \cdots \frac{2}{3}, \text { 当 } n \text { 为大于 } 1 \text { 的正奇数, } I_1=1 .\end{cases}
\end{aligned}
$$
$$
\begin{flalign*}
&(6) \int_0^\pi x f(\sin x) \mathrm{d} x=\frac{\pi}{2} \int_0^\pi f(\sin x) \mathrm{d} x(其中 f(x) 连续).&
\end{flalign*}
$$
